```typescript
{
    "workbench.tree.enableStickyScroll": true, // Enable sticky scroll in the file explorer
    "editor.formatOnSave": true, // Format on save automatically
  	"editor.codeActionsOnSave": {
      "source.fixAll.eslint": "explicit", // Fix all eslint errors on save
      "source.fixAll.stylelint": "explicit" // Fix all stylelint errors on save
  	},
    "editor.guides.bracketPairs": "active", // or true
    "javascript.updateImportsOnFileMove.enabled": "always",
		"typescript.updateImportsOnFileMove.enabled": "always",
}
```
